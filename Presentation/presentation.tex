\section{Introduzione Teorica}
% ===================================================

% ---------------------------------------------------
\begin{frame}{Il Paradigma Moderno: Deep Double Descent}
    \begin{columns}
        \begin{column}{0.5\textwidth}
            \textbf{La Teoria Classica:}
            \begin{itemize}
                \item Trade-off Bias-Varianza a forma di ``U''.
                \item Modelli complessi memorizzano il rumore (``overfitting'').
            \end{itemize}
            \vspace{0.3cm}
            \textbf{La Realtà del Deep Learning:}
            \begin{itemize}
                \item Superata la \textbf{soglia di interpolazione}, l'errore di test torna a scendere.
                \item Modelli massivamente sovra-parametrizzati generalizzano in modo eccellente.
            \end{itemize}
        \end{column}

% ---------------------------------------------------
\begin{frame}{Cos'è il Neural Collapse (NC)?}
    Nelle fasi terminali dell'addestramento, le rappresentazioni dell'ultimo strato mostrano 4 proprietà geometriche (Papyan et al., 2020):
    \vspace{0.3cm}
    \begin{enumerate}
        \item \textbf{NC1 (Variability Collapse):} Le attivazioni convergono verso il loro centro di classe (varianza intra-classe $\to 0$).
        \item \textbf{NC2 (Simplex ETF):} I centri delle classi si dispongono simmetricamente, massimizzando la distanza (Equiangular Tight Frame).
        \item \textbf{NC3 (Self-Duality):} I pesi del classificatore lineare si allineano perfettamente ai centri di classe.
        \item \textbf{NC4 (NCC):} Le previsioni della rete coincidono con una classificazione Nearest Class Center.
    \end{enumerate}
\end{frame}

\section{Setup Sperimentale}
% ===================================================

% ---------------------------------------------------
\begin{frame}{Setup Sperimentale}
    \textbf{Obiettivo:} Unire empiricamente Double Descent, Neural Collapse e analisi delle similarità tra strati (CKA).
    
    \vspace{0.3cm}
    \begin{itemize}
        \item \textbf{Dataset:} MNIST-1D. Dati a bassa dimensionalità ma morfologicamente complessi.
        \item \textbf{Rumore introdotto:} \textcolor{red}{\textbf{20\% Label Noise}}. Forza la rete a memorizzare etichette errate, esacerbando il picco di interpolazione.
        \item \textbf{Architetture:} \texttt{ResNet1D} e \texttt{StandardCNN1D}.
        \item \textbf{Scaling:} Larghezza variabile da $W=2$ a $W=64$ ($10^2 \to 10^5$ param).
        \item \textbf{Ottimizzatori:} SGD vs Adam.
    \end{itemize}
\end{frame}

\begin{frame}{Risultati: La Rottura del Collapse nel Regime Critico}
    \begin{columns}
        \begin{column}{0.5\textwidth}
            \begin{itemize}
                \item L'errore di generalizzazione correla perfettamente con la metrica \textbf{NC1}.
                \item Attorno ai $10^3-10^4$ parametri, la varianza intra-classe esplode. Il modello rompe la sua geometria per memorizzare il rumore.
                \item Superata la soglia ($>10^4$), il Neural Collapse si ripristina e l'errore crolla.
            \end{itemize}

\begin{frame}{L'Impatto dell'Ottimizzatore: SGD domina Adam}
    \begin{columns}
        \begin{column}{0.5\textwidth}
            \begin{itemize}
                \item Nel regime ``overparameterized'', \textbf{SGD} raggiunge configurazioni geometriche (NC2) molto migliori.
                \item \textbf{Adam} tende a bloccarsi in minimi locali sub-ottimali dal punto di vista geometrico.
                \item \textbf{Perché?} SGD possiede un \textit{bias induttivo implicito} che penalizza la norma dei pesi e favorisce margini ampi e simmetrici.
            \end{itemize}
        \end{column}

\begin{frame}{I Limiti Dimensionali del Neural Collapse}
    \begin{columns}
        \begin{column}{0.5\textwidth}
            \begin{itemize}
                \item Matematicamente, un Simplex ETF perfetto con $K$ classi richiede una dimensione latente \textbf{$d \ge K-1$}.
                \item Per MNIST-1D ($K=10$), ci aspettiamo il collasso a partire da $d=9$.
                \item \textbf{Verifica Empirica:} NC2 crolla verticalmente e si stabilizza a zero esattamente in corrispondenza del limite teorico.
            \end{itemize}

    \begin{columns}
        \begin{column}{0.5\textwidth}
            \textbf{CKA:} Misura la similarità geometrica ($0 \to 1$).
            \vspace{0.2cm}
            \begin{itemize}
                \item \textbf{L0/L1:} CKA elevato. Reti critiche e sovra-parametrizzate estraggono le stesse feature a basso livello.

\begin{frame}{CKA: Riproducibilità e Degenerazione Geometrica}
    \begin{columns}
        \begin{column}{0.5\textwidth}
            Stesso modello ($W=32$), solo \textit{Seed casuali iniziali diversi}:
            \vspace{0.2cm}
            \begin{itemize}
                \item L0 apprende feature deterministiche ($CKA = 0.97$).
                \item \textbf{L3 varia molto ($CKA = 0.59$).}
                \item \textbf{Degenerazione Geometrica:} Esistono infinite rotazioni ortogonali per creare un Simplex perfetto. Le reti convergono a soluzioni logicamente isomorfe, ma in coordinate assolute diverse.

\section{Conclusioni}
% ===================================================

% ---------------------------------------------------
\begin{frame}{Conclusioni}
    \begin{enumerate}
        \item \textbf{Il legame Empirico:} Il fenomeno del Double Descent è intimamente legato alla rottura (nel regime critico) e successiva ricostruzione (nel regime massivo) del Neural Collapse.
        \item \textbf{Ottimizzazione:} SGD domina Adam, fungendo da regolarizzatore implicito che guida le reti verso un Simplex ETF perfetto.
        \item \textbf{Geometria Profonda:} Il CKA ha rivelato la disomogeneità dell'apprendimento. Mentre i primi strati sono statici e deterministici, l'ultimo strato degenera geometricamente compiendo una lenta riorganizzazione per massimizzare i margini.
    \end{enumerate}
\end{frame}
